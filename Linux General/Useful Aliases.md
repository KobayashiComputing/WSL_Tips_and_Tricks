# Useful Aliases
```
alias df='df -h'
alias diff='diff --color=auto'
alias du='du -chs'
alias egrep='egrep --color=auto'
alias fgrep='fgrep --color=auto'
alias firefox='firefox &>/dev/null &'
alias grep='grep --color=auto'
alias ip='ip --color=auto'
alias l='ls -CF'
alias la='ls -A'
alias ll='ls -l'
alias ls='ls -Alh --color=auto'
```
